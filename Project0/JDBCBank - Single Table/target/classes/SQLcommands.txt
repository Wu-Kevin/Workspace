create or replace procedure add_user (username varchar2, password varchar2) as
begin
    insert into accounts values(
        userid_generator.nextval,
        accountid_generator.nextval,
        username,
        password,
        0);
end;

create or replace procedure add_account (userID number, username varchar2, password varchar2) as
begin
    insert into accounts values(
        userID,
        accountid_generator.nextval,
        username,
        password,
        0);
end;
